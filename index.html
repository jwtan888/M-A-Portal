async function addEntry() { const m = document.getElementById('mID').value; const pic = document.getElementById('picName').value; const f = document.getElementById('fName').value; const t = document.getElementById('sTime').value; const isNewRequest = document.getElementById('requestToggle').checked; if(!pic || !f || !t) return alert("Please fill in PIC, File Name, and Time."); const entry = { timestamp: new Date().toLocaleString(), machine: m, pic: pic, type: isNewRequest ? "NEW REQUEST" : "SETTING", file: f, time: t.toString() }; const btn = document.getElementById('submitBtn'); btn.innerText = "Syncing..."; btn.disabled = true; try { // We add "mode: 'no-cors'" to bypass browser security blocks await fetch(powerAutomateURL, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(entry) }); // With no-cors, we assume success if no network error occurs allRecords.unshift(entry); localStorage.setItem('factory_logs_v4', JSON.stringify(allRecords)); document.getElementById('fName').value = ''; document.getElementById('sTime').value = ''; document.getElementById('requestToggle').checked = false; applyFilters(); alert("Record sent to Excel!"); } catch (error) { console.error(error); alert("Network Error: Check your internet connection."); } finally { btn.innerText = "Add Record"; btn.disabled = false; } }
